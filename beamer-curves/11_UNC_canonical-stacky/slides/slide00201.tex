\begin{frame}{Log canonical map/ring}


  \begin{definition}
    The \textbf{log} canonical map $C \hookrightarrow \P^{g-1}$ is given by $P
    \mapsto [\omega_1(P):\ldots:\omega_g(P)]$.
  \end{definition}

  \begin{example}
    $\P^1$ and rational normal curve
  \end{example}

  \begin{definition}
    \textbf{log canonical ring}
  \end{definition}




\end{frame}
